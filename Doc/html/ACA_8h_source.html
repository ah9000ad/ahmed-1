00036 bool ACA_row_step(MATGEN_T& MatGen, unsigned b1, unsigned n1, unsigned b2, unsigned n2,
00037                   unsigned& klast, unsigned& i0, unsigned k, unsigned& no, int* Z, int* S,
00038                   T* U, T* V, typename num_traits<T>::abs_type& nrmlsk2, typename num_traits<T>::abs_type scale)
00039 {
00040   typedef typename num_traits<T>::abs_type abs_T;
00041   unsigned l, j0;
00042   abs_T nrmu2, nrmv2, absmax;
00043   T *const pu = U + k*n1, *const pv = V + k*n2;
00044 
00045   // find next non-zero row, start with i0
00046   do {
00047     MatGen.cmpbl(b1+i0, 1, b2, n2, pv);
00048     blas::conj(n2, pv);
00049 
00050     // subtract previously generated outer products from this row
00051     for (l=0; l<k; ++l) {
00052       const T d = nconj(U[l*n1+i0]);
00053       blas::axpy(n2, d, V + l*n2, pv);
00054     }
00055 
00056     // find maximum
00057     absmax = 0.0;
00058     for (l=0; l<n2; ++l) {
00059       if (S[l]>=0) {
00060         const abs_T eabs = abs(pv[l]);
00061         if (eabs>absmax) {
00062           absmax = eabs;
00063           j0 = l;
00064         }
00065       }
00066     }
00067 
00068     // pivot too small ? get new row
00069     if (absmax < 1e-14 * scale) {
00070       //      std::cout << "pivot too small " << no+1 << ' ' << n1 << std::endl;
00071       Z[i0] = -1;
00072       if (++no<n1) i0 = nexti(n1, Z, i0);
00073     }
00074   } while (absmax < 1e-14 * scale && no<n1);
00075 
00076   if (absmax < 1e-14 * scale) return false;    // remainder is zero, return
00077 
00078   // else: we have found a reasonable pivot (i0,j0)
00079   //  std::cout << "i0=" << i0 << ", j0=" << j0 << std::endl;
00080   abs_T sqrtpiv = sqrt(absmax);
00081   nrmv2 = 0.0;
00082   T sca = sqrtpiv/pv[j0];        // scale \tilde{v}_{k+1} -> v_{k+1}
00083   for (l=0; l<n2; ++l) {
00084     if (S[l]>=0) {
00085       const T e = (pv[l] *= sca);
00086       nrmv2 += abs2(e);
00087       if (abs(e) > 1e-8*scale*sqrtpiv) ++S[l];  // col l will be succ. approx
00088     }
00089   }
00090 
00091   MatGen.cmpbl(b1, n1, b2+j0, 1, pu);
00092 
00093   for (l=0; l<k; ++l) {
00094     const T d = nconj(V[l*n2 + j0]);
00095     blas::axpy(n1, d, U + l*n1, pu);
00096   }
00097 
00098   klast = Z[i0];
00099   Z[i0] = S[j0] = -1;
00100   ++no;
00101 
00102   abs_T dsca = 1.0/sqrtpiv;
00103 
00104   absmax = nrmu2 = 0.0;                 // compute norm of u_{k+1}, new i0
00105   for (l=0; l<n1; ++l) {
00106     const abs_T eabs = abs(pu[l] *= dsca);
00107     if (Z[l]>=0 && eabs > 1e-8*scale*sqrtpiv) ++Z[l];
00108     nrmu2 += eabs*eabs;
00109 
00110     // find new pivot i0
00111     if (Z[l]>=0 && eabs>absmax) {
00112       i0 = l;
00113       absmax = eabs;
00114     }
00115   }
00116 
00117   nrmlsk2 = nrmu2 * nrmv2;
00118   return true;
00119 }

00212   // klast is the number of successful steps applied to the last pivotal row
00213   int kmin = -2, keps = -1;
00214   abs_T nrmlsk2, nrms2 = 0.0;
00215   T sum;

00217

00219 
00220   U = new T[(kmax+1)*n1];
00221   V = new T[(kmax+1)*n2];
00222   assert(U!=NULL && V!=NULL);
00223 
00224   int *Z = new int[n1], *S = new int[n2];
00225   assert(Z!=NULL && S!=NULL);
00226 
00227   for (l=0; l<n1; ++l) Z[l] = 0;
00228   for (l=0; l<n2; ++l) S[l] = 0;
00229 
00230   k = 0;
00231 
00232   do {
00233 
00234     // compute a cross
00235     inext = i0;
00236     ok = ACA_row_step(MatGen, b1, n1, b2, n2, klast, inext, k, no, Z, S,

00240     if (!ok) { // in last step no non-zero row could be found
00241       nrmlsk2 = 0.0;
00242       goto SUCC;
00243     }
00244 
00245     // else: step was successful
00246 
00247     // ------------------------------------------------------------------------
00248     // check stopping criterion
00249 
00250     sum = (T) 0.0;                            // update nrms2
00251     for (l=0; l<k; ++l)
00252       sum += blas::scpr(n1, U+l*n1, U+k*n1) * blas::scpr(n2, V+l*n2, V+k*n2);
00253 
00254     nrms2 += 2.0 * Re(sum) + nrmlsk2;
00255 
00256     stpcrit = (nrmlsk2<eps*eps*nrms2);
00257     if (stpcrit && keps==-1) keps = klast+1;  // keps is the rank for acc. eps
00258

00260 
00261     ++k;
00262 
00263     // adjust scale (estimated entry size of the next remainder)
00264     scale = sqrt(nrmlsk2/(n1*n2));
00265

00267     // find row with smallest number of succ. subtractions
00268

00270 
00271       //      for (unsigned j=0; j<n1; ++j) std::cout << Z[j] << ' ';
00272       //      std::cout << std::endl;
00273 
00274       for (i0=0; i0<n1 && Z[i0]<0; ++i0);
00275       if (i0==n1) {
00276         nrmlsk2 = 0.0;
00277         goto SUCC;  // last step was the last possible step
00278       }
00279 
00280       // else: Z[i0]>=0
00281       kmin = Z[i0];
00282       for (unsigned i=i0+1; i<n1; ++i)
00283         if (Z[i]>=0 && Z[i]<kmin) kmin = Z[i0=i];
00284 
00285       if ((unsigned)kmin==k) i0 = inext;
00286     } else i0 = inext; // use max of last column as new pivot
00287 
00288     //    std::cout << "kmin=" << kmin << ", keps=" << keps << std::endl;
00289   } while (no<n1 && k<kmax && kmin<keps);
00290 
00291   //  for (unsigned j=0; j<n2; ++j) std::cout << S[j] << ' ';
00292   //  std::cout << std::endl;
00293 
00294   // -----------------------------------------------------------------------
00295   // continue with column oriented ACA
00296 
00297   if ((no==n1 || k==kmax) && kmin<keps) goto NSUCC;
00298 
00299   // guarantee kmin>=keps for the columns, too
00300   do {
00301     for (j0=0; j0<n2 && S[j0]<0; ++j0);
00302     if (j0==n2) {
00303       nrmlsk2 = 0.0;
00304       goto SUCC;              // last step was the last possible step
00305     }
00306

00309     for (unsigned j=j0+1; j<n2; ++j)
00310       if (S[j]>=0 && S[j]<kmin) kmin = S[j0=j];

00312     if (kmin>=keps) goto SUCC;   // if all were succ., stop iteration
00313

00318       nrmlsk2 = 0.0;

00321     ++k;    // next rank-1 approximation
00322

00330 
00331 SUCC:
00332   double epsa = sqrt(nrmlsk2/nrms2);
00333   //  std::cout << "SUCC: " << epsa << ' ' << eps << std::endl;

00335
